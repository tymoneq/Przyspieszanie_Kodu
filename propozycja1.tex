\subsection{Dalsze przykłady}
\begin{frame}{Matematyka}
  % Są predefiniowane bloki na definicje, twierdzenia, przykłady...
  \begin{definition}
  \textit{Błędem średniokwadratowym} estymatora $\hat{\theta}$ parametru $\theta$ nazywamy wielkość
  $$\operatorname{MSE} ({\hat {\theta }})= \mathbb{E} [({\hat {\theta }}-\theta )^{2}].$$
  \end{definition}
  
  % Można też po prostu wstawić blok z nazwą.
  \begin{block}{Wzorek}
    \[
      \hat{\theta} = \arg\min_{\theta} \sum_{i=1}^{n} \ell\big(y_i, f_{\theta}(x_i)\big)
    \]
  \end{block}
  
% linijka matematycznych wzorów na kilka sposobów:
  $$ 2 \cdot 2 = 4 $$
  \[ 2 + 2 = 4 \]
 \begin{equation}
     2^2 = 4
 \end{equation}
  \begin{equation*}
     2 - (-2) = 4
 \end{equation*}
\end{frame}
